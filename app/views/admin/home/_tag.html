如何添加标签(TAG)

显示某个页面的标签

     <% @page.tag_list.each do |tag| %> 
       <%= link_to tag, tag_path(tag) %>
     <% end %>

显示标签云

    <% tag_cloud(@tags, %w(css1 css2 css3 css4)) do |tag, css_class| %>
       <li><%= link_to tag_path(tag.name)  do%>
           <i class="icon-tags"></i><%= tag.name %>
       <% end %></li>
    <% end %>
