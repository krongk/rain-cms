设置参数／常量

参数／常量分为两类

  1. 在application.yml 中设置的程序常量 
    #Use for Mail/SiteMap
DOMAIN_NAME: www.zhidak.com
HOST_NAME: http://www.zhidak.com

#Use for Mail
GMAIL_USERNAME: kenrome
GMAIL_PASSWORD: www.zhidak.com

#Use for init accoutn on db/seeds.rb
ADMIN_NAME: master
ADMIN_EMAIL: master@inruby.com
ADMIN_PASSWORD: kenxxx
ROLES: [admin, user, VIP]

#Use for send sms
SMS_TOGGLE: true
SMS_BAO_USER: inruby
SMS_BAO_PASSWORD: kenxx
#SMS_PHONE: 18080810818|18080810818
SMS_PHONE: 18080810818

#product e-commerce toggle
# 如果开通电子商务，则会显示产品价格等字段
PRODUCT_TOGGLE: true
# 如果开通分销开关，则后台可获得总部数据
FENXIAO_TOGGLE: true
DB_HOST: localhost
DB_PORT: 3306
DB_DATABASE: zhidak
DB_USERNAME: root
DB_PASSWORD: kenrome001
HEADPARTER_SITE: http://www.zhidak.com
